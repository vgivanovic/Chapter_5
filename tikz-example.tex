\documentclass[letterpaper,oneside,article,12pt,oldfontcommands]{memoir}

\usepackage{fontspec}
\setmainfont[Ligatures=TeX,Numbers=OldStyle,Numbers=Monospaced]{Alegreya}
\setsansfont[Ligatures=TeX,Numbers=OldStyle,Numbers=Monospaced]{Alegreya Sans}
\setmonofont[Scale=0.8]{Fira Mono}
\setmathtt[Scale=MatchLowercase]{Fira Math}

\usepackage{tikz}[align=flush left]
\usetikzlibrary {arrows,calc,positioning,shapes.misc}
\tikzset{terminal/.style={
    rectangle,minimum size=6mm,rounded corners=2mm,
    very thick,draw=black!50,
    top color=white,bottom color=black!20,
    font=\normalfont}}
\tikzset{every picture/.style={line width=2pt}}

\begin{document}

\noindent{}
\begin{tikzpicture}[node distance=20mm]
                   
\node (claim)   [terminal,text width=6cm]
                {\textsc{\textbf{Claim}}\\ The statement or conclusion to be verified.};
\node (grounds) [terminal, right=of claim, text width=6cm]
                {\textsc{\textbf{Ground}}\\ The evidence that provides the basis for making the claim};
\node (warrant) [terminal, below=of claim, text width=6cm]
                {\textsc{\textbf{Warrant}}\\ Connect the claims to the evidence.};
\node (qualifier) [terminal,right=of warrant, text width=6cm]
                  {\textsc{\textbf{Qualifier}}\\ The degree of certainty about the claim.};
\node (backing) [terminal, below=of warrant, text width=6cm]
                {\textsc{\textbf{Backing}}\\ Buttresses the warrant.};
\node (rebuttal) [terminal, right=of backing, text width=6cm]
                {\textsc{\textbf{Rebuttal}}\\ Counter-arguments to objections.};

\draw (claim.east) edge[-latex] node[font=\tiny,above]{supported by}    (grounds.west);

\draw [-latex]  (warrant.north)   -- (claim.south);
\draw (warrant.north) edge[-latex] node[font=\tiny,above]{connected by}  (backing.north);
\draw [-latex] (warrant)   -- (rebuttal);

\end{tikzpicture}

\end{document}
